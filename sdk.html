using System;
using System.Net.Http;
using System.Text;
using System.Text.Json;
using System.Threading.Tasks;
using System.Management;
using System.Security.Cryptography;

public class NexAuth
{
    private static readonly HttpClient http = new HttpClient();
    private const string API = "https://nexauth-backend.onrender.com/api";
    private string appId;

    public bool IsValid { get; private set; }
    public string LicenseType { get; private set; }

    public NexAuth(string appId) { this.appId = appId; }

    public static string GetHWID()
    {
        try {
            string raw = "";
            using (var s = new ManagementObjectSearcher("SELECT ProcessorId FROM Win32_Processor"))
                foreach (var o in s.Get()) raw += o["ProcessorId"];
            using (var sha = SHA256.Create()) {
                var bytes = sha.ComputeHash(Encoding.UTF8.GetBytes(raw));
                return BitConverter.ToString(bytes).Replace("-", "").Substring(0, 32);
            }
        } catch { return Environment.MachineName; }
    }

    public async Task<bool> ValidateLicense(string licenseKey)
    {
        var body = JsonSerializer.Serialize(new { appId, key = licenseKey, hwid = GetHWID() });
        var res = await http.PostAsync(API + "/licenses/validate",
            new StringContent(body, Encoding.UTF8, "application/json"));
        var json = JsonDocument.Parse(await res.Content.ReadAsStringAsync()).RootElement;
        IsValid = json.GetProperty("success").GetBoolean();
        if (IsValid) LicenseType = json.GetProperty("license").GetProperty("type").GetString();
        return IsValid;
    }

    public async Task<bool> Login(string username, string password)
    {
        var body = JsonSerializer.Serialize(new { username, password });
        var res = await http.PostAsync(API + "/auth/login",
            new StringContent(body, Encoding.UTF8, "application/json"));
        var json = JsonDocument.Parse(await res.Content.ReadAsStringAsync()).RootElement;
        return json.GetProperty("success").GetBoolean();
    }
}

        // Windows only
        char buf[256];
        FILE* f = _popen("wmic cpu get ProcessorId /value", "r");
        std::string result;
        while (fgets(buf, sizeof(buf), f)) result += buf;
        _pclose(f);
        return result.substr(0, 32);

        json body = { {"username", username}, {"password", password} };
        std::string resp = Post("/auth/login", body.dump());

import requests
import hashlib
import subprocess
import platform

class NexAuth:
    API = "https://nexauth-backend.onrender.com/api"

    def __init__(self, app_id):
        self.app_id = app_id
        self.is_valid = False
        self.license_type = None

    def get_hwid(self):
        try:
            if platform.system() == "Windows":
                out = subprocess.check_output(
                    "wmic cpu get ProcessorId", shell=True
                ).decode()
            else:
                out = str(platform.node())
            return hashlib.sha256(out.encode()).hexdigest()[:32]
        except:
            return platform.node()

    def validate_license(self, license_key):
        res = requests.post(self.API + "/licenses/validate", json={
            "appId": self.app_id,
            "key": license_key,
            "hwid": self.get_hwid()
        })
        data = res.json()
        self.is_valid = data.get("success", False)
        if self.is_valid:
            self.license_type = data["license"]["type"]
        return self.is_valid

    def login(self, username, password):
        res = requests.post(self.API + "/auth/login", json={
            "username": username,
            "password": password
        })
        data = res.json()
        return data.get("success", False), data.get("token")

import com.google.gson.*;
import java.net.*;
import java.net.http.*;
import java.security.*;
import java.nio.charset.StandardCharsets;

public class NexAuth {
    private static final String API = "https://nexauth-backend.onrender.com/api";
    private final String appId;
    public boolean isValid = false;
    public String licenseType;
    private final HttpClient client = HttpClient.newHttpClient();
    private final Gson gson = new Gson();

    public NexAuth(String appId) { this.appId = appId; }

    public String getHWID() {
        try {
            String raw = System.getenv("COMPUTERNAME") + System.getenv("USERNAME");
            MessageDigest md = MessageDigest.getInstance("SHA-256");
            byte[] hash = md.digest(raw.getBytes(StandardCharsets.UTF_8));
            StringBuilder sb = new StringBuilder();
            for (byte b : hash) sb.append(String.format("%02x", b));
            return sb.toString().substring(0, 32);
        } catch (Exception e) { return "java-hwid"; }
    }

    public boolean validateLicense(String key) throws Exception {
        JsonObject body = new JsonObject();
        body.addProperty("appId", appId);
        body.addProperty("key", key);
        body.addProperty("hwid", getHWID());
        String resp = post("/licenses/validate", body.toString());
        JsonObject r = gson.fromJson(resp, JsonObject.class);
        isValid = r.get("success").getAsBoolean();
        if (isValid) licenseType = r.getAsJsonObject("license").get("type").getAsString();
        return isValid;
    }

    public boolean login(String username, String password) throws Exception {
        JsonObject body = new JsonObject();
        body.addProperty("username", username);
        body.addProperty("password", password);
        String resp = post("/auth/login", body.toString());
        return gson.fromJson(resp, JsonObject.class).get("success").getAsBoolean();
    }

    private String post(String path, String body) throws Exception {
        HttpRequest req = HttpRequest.newBuilder()
            .uri(URI.create(API + path))
            .header("Content-Type", "application/json")
            .POST(HttpRequest.BodyPublishers.ofString(body))
            .build();
        return client.send(req, HttpResponse.BodyHandlers.ofString()).body();
    }
}

JavaScript / Node.js — NexAuth SDK
Works with Electron, Node.js apps

nexauth.js — Drop in your project
const https = require('https');
const crypto = require('crypto');
const os = require('os');
const { execSync } = require('child_process');

class NexAuth {
    constructor(appId) {
        this.appId = appId;
        this.isValid = false;
        this.licenseType = null;
        this.API = 'https://nexauth-backend.onrender.com/api';
    }

    getHWID() {
        try {
            let raw = os.hostname() + os.cpus()[0].model;
            return crypto.createHash('sha256').update(raw).digest('hex').slice(0, 32);
        } catch { return os.hostname(); }
    }

    async post(path, body) {
        const data = JSON.stringify(body);
        return new Promise((resolve, reject) => {
            const req = https.request({
                hostname: 'nexauth-backend.onrender.com',
                path: '/api' + path, method: 'POST',
                headers: { 'Content-Type': 'application/json', 'Content-Length': data.length }
            }, res => {
                let body = '';
                res.on('data', chunk => body += chunk);
                res.on('end', () => resolve(JSON.parse(body)));
            });
            req.on('error', reject);
            req.write(data); req.end();
        });
    }

    async validateLicense(key) {
        const r = await this.post('/licenses/validate',
            { appId: this.appId, key, hwid: this.getHWID() });
        this.isValid = r.success;
        if (r.success) this.licenseType = r.license.type;
        return r.success;
    }

    async login(username, password) {
        const r = await this.post('/auth/login', { username, password });
        return { success: r.success, token: r.token };
    }
}

module.exports = NexAuth;

NexAuth.vb — Add to your project
Imports System.Net.Http
Imports System.Text
Imports System.Text.Json
Imports System.Management
Imports System.Security.Cryptography

Public Class NexAuth
    Private Shared http As New HttpClient()
    Private Const API As String = "https://nexauth-backend.onrender.com/api"
    Private appId As String

    Public IsValid As Boolean = False
    Public LicenseType As String

    Public Sub New(appId As String)
        Me.appId = appId
    End Sub

    Public Shared Function GetHWID() As String
        Try
            Dim raw As String = ""
            Dim searcher As New ManagementObjectSearcher("SELECT ProcessorId FROM Win32_Processor")
            For Each obj In searcher.Get()
                raw &= obj("ProcessorId").ToString()
            Next
            Using sha = SHA256.Create()
                Dim bytes = sha.ComputeHash(Encoding.UTF8.GetBytes(raw))
                Return BitConverter.ToString(bytes).Replace("-", "").Substring(0, 32)
            End Using
        Catch
            Return Environment.MachineName
        End Try
    End Function

    Public Async Function ValidateLicense(key As String) As Task(Of Boolean)
        Dim body = $"{{""appId"":""{appId}"",""key"":""{key}"",""hwid"":""{GetHWID()}""}}"
        Dim content = New StringContent(body, Encoding.UTF8, "application/json")
        Dim res = Await http.PostAsync(API + "/licenses/validate", content)
        Dim json = JsonDocument.Parse(Await res.Content.ReadAsStringAsync()).RootElement
        IsValid = json.GetProperty("success").GetBoolean()
        If IsValid Then LicenseType = json.GetProperty("license").GetProperty("type").GetString()
        Return IsValid
    End Function

    Public Async Function Login(username As String, password As String) As Task(Of Boolean)
        Dim body = $"{{""username"":""{username}"",""password"":""{password}""}}"
        Dim content = New StringContent(body, Encoding.UTF8, "application/json")
        Dim res = Await http.PostAsync(API + "/auth/login", content)
        Dim json = JsonDocument.Parse(Await res.Content.ReadAsStringAsync()).RootElement
        Return json.GetProperty("success").GetBoolean()
    End Function
End Class
